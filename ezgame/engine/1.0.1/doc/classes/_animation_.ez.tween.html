Class Tween

缓动动画

example 
//json模式
var tween = new Tween(this.icon, [
    { move: [ { x: [10, 100], y: [30, 300] }, 1000] },
    { to: [ { x: 30 , y: 0 } , 500] },
    { wait: 500 },
    { set: { x: 10,  y: 30 } },
    { call: [this.onFinish, this] },
    { target: target }
])
//调用链模式
var tween = new egl.Tween(icon)
    .move({ x: [10, 100], y: [30, 300] }, 1000)
    .to({ x: 30, y: 0 }, 500)
    .wait(500)
    .set({ x: 10, y: 30 })
    .call(onFinish).play();

  • func(duration: number, func: function, thisObj?: any, ease?: function): Tween

  • 用自定义函数设置动画

    example 
    //让目标对象绕圈转动
new Tween(arrow).func(5000, (target, t) => {
        target.x = Math.cos(t * Math.PI * 2) * 100;
        target.y = Math.sin(t * Math.PI * 2) * 100;
        target.angle = t * 360;
}).config({ loop: true }).play();
